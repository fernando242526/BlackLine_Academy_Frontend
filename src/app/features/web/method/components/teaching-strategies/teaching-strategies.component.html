def challenge_based_learning(problem, skills, time_limit):
    solution = analyze_problem(problem)
    while not is_solved(problem, solution) and time_limit > 0:
        new_approach = apply_skills(skills, problem)
        solution = implement(new_approach)
        feedback = evaluate(solution)
        solution = refine(solution, feedback)
        time_limit -= 1
    return solution, learn_from_process(problem, solution)
